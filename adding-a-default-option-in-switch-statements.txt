function switchOfStuff(val) {
  var answer = "";
  // Only change code below this line
  switch (val) {
    case "a":
      return "apple";
      break;
    case "b":
      return "bird";
      break;
    case "c":
      return "cat";
      break;
    case "d":
      return "stuff";
      break;
    case 4:
      return "stuff";
      break;
    default:
      answer = "stuff";
  }


  // Only change code above this line
  return answer;
}

switchOfStuff(1);
